CREATE TABLE CURRENCY (
    CURRENCY_CODE       VARCHAR2(3),  
    CURRENCY_NAME       VARCHAR2(30),
    USD_EXCHANGE_RATE   NUMBER(12,6),  

CONSTRAINT PK_CURRENCY_CODE PRIMARY KEY (CURRENCY_CODE),
CONSTRAINT NN_CURRENCY_NAME CHECK (CURRENCY_NAME IS NOT NULL),
CONSTRAINT NN_USD_EXCHANGE_RATE CHECK (USD_EXCHANGE_RATE IS NOT NULL)  
);  


CREATE TABLE COUNTRY (
    ISO_CODE        VARCHAR2(3),
    COUNTRY_NAME    VARCHAR2(80),
    CURRENCY_CODE   VARCHAR2(3), 

CONSTRAINT PK_COUNTRY_CODE PRIMARY KEY (ISO_CODE),
CONSTRAINT NN_COUNTRY_NAME CHECK (COUNTRY_NAME IS NOT NULL),
CONSTRAINT FK_COUNTRY_CURRENCY FOREIGN KEY (CURRENCY_CODE) REFERENCES CURRENCY (CURRENCY_CODE)
);
  

CREATE TABLE CITY (  
    CITY_ID NUMBER GENERATED BY DEFAULT AS IDENTITY,
    CITY_NAME   VARCHAR2(50),
    ISO_CODE    VARCHAR2(3), 

CONSTRAINT PK_CITY_ID PRIMARY KEY (CITY_ID),
CONSTRAINT NN_CITY_NAME CHECK (CITY_NAME IS NOT NULL),
CONSTRAINT FK_CITY_COUNTRY FOREIGN KEY (ISO_CODE) REFERENCES COUNTRY (ISO_CODE)
);  


CREATE TABLE OFFICE ( 
    OFFICE_ID NUMBER GENERATED BY DEFAULT AS IDENTITY,
    CITY_ID NUMBER,
    ADDRESS VARCHAR2(200),
    TIMEZONE    VARCHAR2(100),

CONSTRAINT PK_OFFICE_ID PRIMARY KEY (OFFICE_ID), 
CONSTRAINT FK_OFFICE_CITY FOREIGN KEY (CITY_ID) REFERENCES CITY (CITY_ID),
CONSTRAINT NN_ADDRESS CHECK (ADDRESS IS NOT NULL), 
CONSTRAINT NN_TIMEZONE CHECK (TIMEZONE IS NOT NULL)
);  


CREATE TABLE US_STATE ( 
    STATE_CODE VARCHAR2(2), 
    STATE_NAME VARCHAR2(20), 
    ISO_CODE   VARCHAR2(3),  

CONSTRAINT PK_STATE_CODE PRIMARY KEY (STATE_CODE), 
CONSTRAINT NN_STATE_NAME CHECK (STATE_NAME IS NOT NULL), 
CONSTRAINT FK_ISO_CODE FOREIGN KEY (ISO_CODE) REFERENCES COUNTRY (ISO_CODE) 
);  


CREATE TABLE US_STATE_TAX_POLICY ( 
    STATE_CODE   VARCHAR2(2), 
    EFFECTIVE_DT   DATE DEFAULT SYSDATE, 
    TAX_RATE  NUMBER(6,4), 

CONSTRAINT PK_STATE_TAX_POL PRIMARY KEY (STATE_CODE, EFFECTIVE_DT)
);  


CREATE TABLE COUNTRY_TAX_POLICY ( 
    ISO_CODE  VARCHAR2(3), 
    EFFECTIVE_DT   DATE DEFAULT SYSDATE,
    TAX_TYPE    VARCHAR2(20) DEFAULT 'GENERAL', 
    TAX_RATE    NUMBER(6,4), 

CONSTRAINT PK_COUNTRY_TAX PRIMARY KEY (ISO_CODE, EFFECTIVE_DT)  
);  


CREATE TABLE ROLE (
    ROLE_ID NUMBER GENERATED BY DEFAULT AS IDENTITY,
    ROLE_NAME VARCHAR2(100) NOT NULL,
    DESCRIPTION VARCHAR2(200), 

CONSTRAINT PK_ROLE PRIMARY KEY (ROLE_ID),
CONSTRAINT UQ_ROLE_NAME UNIQUE (ROLE_NAME)  
);  


CREATE TABLE DEPARTMENT (
    DEPT_ID NUMBER GENERATED BY DEFAULT AS IDENTITY,
    DEPT_NAME   VARCHAR2(50),
    PARENT_DEPT NUMBER, 

CONSTRAINT PK_DEPT_ID PRIMARY KEY (DEPT_ID),
CONSTRAINT FK_PARENT_DEPT FOREIGN KEY (PARENT_DEPT) REFERENCES DEPARTMENT(DEPT_ID)
);  

 
CREATE TABLE TASKS (  
    TASK_ID NUMBER GENERATED ALWAYS AS IDENTITY,
    TASK_NAME VARCHAR2(250),
    TASK_DESC VARCHAR2(500),

CONSTRAINT CONST_PK_TASK_ID PRIMARY KEY (TASK_ID)  
);  


CREATE TABLE WORK_POLICY (   
    POLICY_ID INT GENERATED BY DEFAULT AS IDENTITY,   
    ISO_CODE VARCHAR2(3) NOT NULL,  
    MAX_HOURS_PER_WEEK INT NOT NULL,   
    OVERTIME_RULE VARCHAR2(20),  

CONSTRAINT PK_POLICY_ID PRIMARY KEY (POLICY_ID),  
CONSTRAINT FK_COUNTRY_ID FOREIGN KEY (ISO_CODE) REFERENCES COUNTRY (ISO_CODE)
);  


CREATE TABLE PERSON (
    PERSON_ID NUMBER GENERATED ALWAYS AS IDENTITY,
    P_FNAME VARCHAR2(100) NOT NULL,
    P_LNAME VARCHAR2(100) NOT NULL,
    P_LEGALNAME VARCHAR(100),
    P_BIRTHDATE DATE NOT NULL,
    P_EMAIL VARCHAR2(100),  
    P_PHONE NUMBER,  
    P_STATUS VARCHAR(20) DEFAULT 'Active' NOT NULL, 
    P_ROLE  VARCHAR(1) CHECK (P_ROLE IN ('E', 'C')),  

CONSTRAINT CONST_PK_PERSON_ID PRIMARY KEY (PERSON_ID)  
);  

 
CREATE TABLE PROSPECT (
    PROSPECT_ID NUMBER GENERATED ALWAYS AS IDENTITY,   
    TYPICAL_NAME VARCHAR2(100) NOT NULL,  
    LEGAL_NAME VARCHAR2(100)NOT NULL,   
    EXTERNAL_ID VARCHAR2(80) NOT NULL,   
    FK_COUNTRY_ID VARCHAR2(3),  
    PROSP_STATUS VARCHAR2(20) DEFAULT 'Active' NOT NULL,   
    CREATE_DATE TIMESTAMP DEFAULT CURRENT_TIMESTAMP NOT NULL,   

CONSTRAINT CONST_PK_PROSPECT_ID PRIMARY KEY (PROSPECT_ID),  
CONSTRAINT CONST_FK_COUNTRY_ID FOREIGN KEY (FK_COUNTRY_ID) REFERENCES COUNTRY(ISO_CODE)  
);  


CREATE TABLE client (  
    Client_id NUMBER GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,   
    prospect_id NUMBER NOT NULL,   
    start_date  DATE NOT NULL,  
    active_status CHAR(1) DEFAULT 'Y' NOT NULL,  

CONSTRAINT FK_CLIENT_PROSPECT  FOREIGN KEY (PROSPECT_ID) REFERENCES PROSPECT (PROSPECT_ID),    
CONSTRAINT ck_client_active_status CHECK (active_status IN ('Y','N'))  
);  


CREATE TABLE client_division (  
    division_id NUMBER GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,  
    client_id   NUMBER NOT NULL,
    cl_name VARCHAR2(100) NOT NULL,  
    parent_division_id NUMBER NULL,  
    active_status   CHAR(1) DEFAULT 'Y' NOT NULL,  

CONSTRAINT ck_div_active CHECK (active_status IN ('Y','N')),  
CONSTRAINT fk_div_client FOREIGN KEY (client_id) REFERENCES client(client_id)  
);  

 
CREATE TABLE EMPLOYEE (  
    EMP_ID NUMBER,  
    HIRE_DATE DATE NOT NULL, END_DATE DATE,  
    EMP_STATUS VARCHAR2(25) DEFAULT 'ACTIVE' NOT NULL,   
    HOME_OFFICE_ID NUMBER,  
    MANAGER_ID NUMBER, DEPT_ID NUMBER,  

CONSTRAINT PK_EMPLOYEE PRIMARY KEY (EMP_ID),  
CONSTRAINT FK_EMPLOYEE_PERSON FOREIGN KEY (EMP_ID) REFERENCES PERSON (PERSON_ID),  
CONSTRAINT FK_EMPLOYEE_OFFICE FOREIGN KEY (HOME_OFFICE_ID) REFERENCES OFFICE (OFFICE_ID),  
CONSTRAINT FK_EMPLOYEE_MANAGER FOREIGN KEY (MANAGER_ID) REFERENCES EMPLOYEE (EMP_ID),  
CONSTRAINT FK_EMPLOYEE_DEPT FOREIGN KEY (DEPT_ID) REFERENCES DEPARTMENT (DEPT_ID),  
CONSTRAINT CK_EMP_STATUS CHECK (EMP_STATUS IN ('ACTIVE','INACTIVE','ON_LEAVE','TERMINATED')),  
CONSTRAINT CK_EMP_DATES CHECK (END_DATE IS NULL OR END_DATE >= HIRE_DATE), 
CONSTRAINT CK_EMP_MANAGER_NE_SELF CHECK (MANAGER_ID IS NULL OR MANAGER_ID <> EMP_ID)  
);  

 
CREATE TABLE EMPLOYEE_ROLE (  
    EMP_ROLE_ID NUMBER GENERATED ALWAYS AS IDENTITY,   
    EMP_ID NUMBER NOT NULL,
    ROLE_ID NUMBER NOT NULL,   
    START_DATE DATE NOT NULL,
    END_DATE DATE,  

CONSTRAINT PK_EMPLOYEE_ROLE PRIMARY KEY (EMP_ROLE_ID),  
CONSTRAINT FK_ER_EMP FOREIGN KEY (EMP_ID) REFERENCES EMPLOYEE (EMP_ID),   
CONSTRAINT FK_ER_ROLE FOREIGN KEY (ROLE_ID) REFERENCES ROLE (ROLE_ID),   
CONSTRAINT CK_ER_DATES CHECK (END_DATE IS NULL OR END_DATE >= START_DATE),   
CONSTRAINT UQ_ER_ONE_AT_START UNIQUE (EMP_ID, ROLE_ID, START_DATE)
);  


CREATE TABLE EMPLOYEE_ASSIGNMENT (  
    ASSIGN_ID INT GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY, 
    EMP_ID INT NOT NULL,
    OFFICE_ID INT NOT NULL,
    START_DATE DATE NOT NULL,   
    END_DATE DATE,  

CONSTRAINT FK_ASSIGN_EMP FOREIGN KEY (EMP_ID) REFERENCES EMPLOYEE (EMP_ID),  
CONSTRAINT FK_ASSIGN_OFFICE FOREIGN KEY (OFFICE_ID) REFERENCES OFFICE (OFFICE_ID)  
);  

 
CREATE TABLE EMPLOYEE_WORK_POLICY (   
    EMP_POLICY_ID INT GENERATED BY DEFAULT AS IDENTITY,
    EMP_ID INT NOT NULL,   
    POLICY_ID INT NOT NULL,  
    EWP_START_DATE DATE NOT NULL,   
    EWP_END_DATE DATE,  

CONSTRAINT PK_EMP_POLICY PRIMARY KEY (EMP_POLICY_ID),  
CONSTRAINT FK_EMP_ID FOREIGN KEY (EMP_ID) REFERENCES EMPLOYEE (EMP_ID),  
CONSTRAINT FK_POLICY_ID FOREIGN KEY (POLICY_ID) REFERENCES WORK_POLICY (POLICY_ID)  
);   

 
CREATE TABLE PROJECT ( 
    PROJECT_ID NUMBER GENERATED BY DEFAULT AS IDENTITY,   
    CLIENT_ID NUMBER NOT NULL,  
    PROJECT_MANAGER NUMBER NOT NULL,  
    PROJECT_NAME VARCHAR2(120) NOT NULL,   
    DESCRIPTION VARCHAR2(200),   
    START_DATE DATE NOT NULL,  
    END_DATE DATE,  
    ACTIVE_STATUS CHAR(1) DEFAULT 'Y' NOT NULL,  

CONSTRAINT PK_PROJECT PRIMARY KEY (PROJECT_ID),  
CONSTRAINT FK_PROJECT_CLIENT FOREIGN KEY (CLIENT_ID) REFERENCES CLIENT (CLIENT_ID), 
CONSTRAINT FK_PROJECT_PM FOREIGN KEY (PROJECT_MANAGER) REFERENCES EMPLOYEE (EMP_ID), 
CONSTRAINT CK_PROJECT_ACTIVE CHECK (ACTIVE_STATUS IN ('Y','N')),  
CONSTRAINT CK_PROJECT_DATES CHECK (END_DATE IS NULL OR END_DATE >= START_DATE)  
);  


CREATE TABLE PROJECT_TEAM ( 
    PROJECT_TEAM_ID NUMBER GENERATED BY DEFAULT AS IDENTITY,   
    PROJECT_ID NUMBER NOT NULL,
    EMP_ID NUMBER NOT NULL,
    ROLE_ID NUMBER,  
    ALLOCATION_PCT NUMBER(5,2) DEFAULT 100 NOT NULL,   
    START_DATE DATE NOT NULL,  
    END_DATE DATE,  

CONSTRAINT PK_PROJECT_TEAM PRIMARY KEY (PROJECT_TEAM_ID),  
CONSTRAINT FK_PT_PROJECT FOREIGN KEY (PROJECT_ID) REFERENCES PROJECT (PROJECT_ID),  
CONSTRAINT FK_PT_EMP FOREIGN KEY (EMP_ID) REFERENCES EMPLOYEE (EMP_ID),   
CONSTRAINT FK_PT_ROLE FOREIGN KEY (ROLE_ID) REFERENCES ROLE (ROLE_ID),   
CONSTRAINT CK_PT_ALLOC CHECK (ALLOCATION_PCT BETWEEN 0 AND 100),   
CONSTRAINT CK_PT_DATES CHECK (END_DATE IS NULL OR END_DATE >= START_DATE),   
CONSTRAINT UQ_PT_ONE_ROW UNIQUE (PROJECT_ID, EMP_ID, START_DATE)  
);  

 
CREATE TABLE PROJECT_TASKS(  
    PROJ_TASK_ID NUMBER GENERATED ALWAYS AS IDENTITY,   
    FK_TASK_ID NUMBER,  
    FK_PROJECT_ID NUMBER,  

CONSTRAINT CONST_PK_PROJ_TASK_ID PRIMARY KEY (PROJ_TASK_ID),  
CONSTRAINT CONST_FK_TASK_ID FOREIGN KEY (FK_TASK_ID) REFERENCES TASKS(TASK_ID), 
CONSTRAINT CONST_FK_PROJECT_ID FOREIGN KEY (FK_PROJECT_ID) REFERENCES PROJECT(PROJECT_ID)
);  

 
CREATE TABLE contract (  
    contract_id NUMBER GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,   
    client_id   NUMBER NOT NULL,
    start_date  DATE NOT NULL,  
    end_date    DATE NOT NULL,   
    billing_method VARCHAR2(20) NOT NULL,  

CONSTRAINT fk_contract_client FOREIGN KEY (client_id) REFERENCES client(client_id),  
CONSTRAINT ck_contract_dates CHECK (start_date <= end_date),  
CONSTRAINT ck_contract_bm CHECK (billing_method IN ('HOURLY', 'FLATRATE'))  
);  

 
CREATE TABLE contracts_hourly (   
    contract_id NUMBER PRIMARY KEY,
    hourly_rate NUMBER(10,2) NOT NULL,  

CONSTRAINT fk_hrly_contract FOREIGN KEY (contract_id) REFERENCES contract(contract_id) ON DELETE CASCADE,  
CONSTRAINT ck_hourly_rate_pos CHECK (hourly_rate > 0)  
);  

 
CREATE TABLE contracts_flatrate (   
    contract_id NUMBER PRIMARY KEY,
    flatrate    NUMBER(12,2) NOT NULL,  

CONSTRAINT fk_flat_contract FOREIGN KEY (contract_id) REFERENCES contract(contract_id) ON DELETE CASCADE,  
CONSTRAINT ck_flat_rate_pos CHECK (flatrate > 0)  
);  

 
CREATE TABLE TIMESHEET (   
    TIMESHEET_ID INT GENERATED BY DEFAULT AS IDENTITY,  
    EMP_ID  INT NOT NULL,   
    WEEK_START_DATE DATE NOT NULL,  
    STATUS  VARCHAR2(20) DEFAULT 'DRAFT' NOT NULL,
    SUBMIT_DATE TIMESTAMP DEFAULT NULL,  
    APPROVAL_DATE   TIMESTAMP DEFAULT NULL,   
    APPROVER_EMP_ID INT,  

CONSTRAINT PK_TIMESHEET PRIMARY KEY (TIMESHEET_ID),  
CONSTRAINT FK_TS_EMP FOREIGN KEY (EMP_ID) REFERENCES EMPLOYEE (EMP_ID),  
CONSTRAINT FK_TS_APPROVER FOREIGN KEY (APPROVER_EMP_ID) REFERENCES EMPLOYEE (EMP_ID)  
);  

 
CREATE TABLE TIME_ENTRY (  
    TIME_ENTRY_ID INT GENERATED BY DEFAULT AS IDENTITY,
    TIMESHEET_ID INT NOT NULL,   
    WORK_DATE DATE NOT NULL,
    HOURS_WORKED NUMBER (6,2) DEFAULT 0.00 NOT NULL,  
    PROJECT_ID INT,  
    CLIENT_ID INT,  
    CONTRACT_ID INT,  
    TASK_CODE INT,  
    NOTES VARCHAR2(100),  

CONSTRAINT PK_TIME_ENTRY PRIMARY KEY (TIME_ENTRY_ID),  
CONSTRAINT FK_TIMESHEET_ID FOREIGN KEY (TIMESHEET_ID) REFERENCES TIMESHEET (TIMESHEET_ID),  
CONSTRAINT FK_PROJECT_ID FOREIGN KEY (PROJECT_ID) REFERENCES PROJECT (PROJECT_ID),  
CONSTRAINT FK_TE_CLIENT_ID FOREIGN KEY (CLIENT_ID) REFERENCES CLIENT (CLIENT_ID),  
CONSTRAINT FK_TE_CONTRACT_ID FOREIGN KEY (CONTRACT_ID) REFERENCES CONTRACT (CONTRACT_ID)  
);  


CREATE TABLE CLIENT_CONTACT(    
    CONTACT_ID NUMBER GENERATED ALWAYS AS IDENTITY,    
    FK_PERSON_ID NUMBER,    
    FK_CLIENT_ID NUMBER,    
    FK_PROSPECT_ID NUMBER,    
    CLIENT_TITLE VARCHAR2(100) NOT NULL,    
    FK_DIVISION_ID NUMBER,    
    IS_BILLING_CONTACT VARCHAR2(1) DEFAULT 'N' NOT NULL CHECK (IS_BILLING_CONTACT IN ('Y', 'N')), 
    IS_SALES_CONTACT VARCHAR2(1) DEFAULT 'N' NOT NULL CHECK (IS_SALES_CONTACT IN ('Y', 'N')),    

CONSTRAINT CONST_PK_CONTACT_ID PRIMARY KEY (CONTACT_ID),    
CONSTRAINT CONST_FK_PERSON_ID FOREIGN KEY (FK_PERSON_ID) REFERENCES PERSON(PERSON_ID),    
CONSTRAINT CONST_FK_CLIENT_ID FOREIGN KEY (FK_CLIENT_ID) REFERENCES CLIENT(CLIENT_ID),    
CONSTRAINT CONST_FK_PROSPECT_ID FOREIGN KEY (FK_PROSPECT_ID) REFERENCES PROSPECT(PROSPECT_ID),    
CONSTRAINT CONST_FK_DIVISION_ID FOREIGN KEY (FK_DIVISION_ID) REFERENCES CLIENT_DIVISION(DIVISION_ID)    
);   

 
CREATE TABLE INVOICE (
    INVOICE_ID NUMBER GENERATED ALWAYS AS IDENTITY,  
    CLIENT_ID NUMBER NOT NULL,
    CONTRACT_ID NUMBER NOT NULL,
    INVOICE_DATE DATE NOT NULL,
    PERIOD_START DATE NOT NULL,  
    PERIOD_END DATE NOT NULL,
    CURRENCY_CODE VARCHAR(3),
    SUBTOTAL DECIMAL(9,2)NOT NULL, 
    TAX DECIMAL(3,2) NOT NULL,  
    TOTAL DECIMAL(9,2) NOT NULL,   
    STATUS VARCHAR2(20) NOT NULL,  

CONSTRAINT PK_INVOICE_ID PRIMARY KEY (INVOICE_ID),  
CONSTRAINT FK_INV_CLIENT_ID FOREIGN KEY(CLIENT_ID) REFERENCES CLIENT (CLIENT_ID),   
CONSTRAINT FK_INV_CONTRACT_ID FOREIGN KEY(CONTRACT_ID) REFERENCES CONTRACT(CONTRACT_ID),  
CONSTRAINT FK_INV_CURRENCY_CODE FOREIGN KEY(CURRENCY_CODE) REFERENCES CURRENCY(CURRENCY_CODE)  
);  


CREATE TABLE INVOICE_LINE (
    LINE_ID NUMBER GENERATED ALWAYS AS IDENTITY NOT NULL,
    INVOICE_ID NUMBER NOT NULL,   
    LINE_DESCRIPTION VARCHAR(200),  
    QTY_HOURS DECIMAL(6,2) NOT NULL,   
    RATE DECIMAL(5,2) NOT NULL, AMOUNT DECIMAL(9,2), 

CONSTRAINT PK_LINE_ID PRIMARY KEY (LINE_ID),  
CONSTRAINT FK_INVOICE_ID FOREIGN KEY(INVOICE_ID) REFERENCES INVOICE (INVOICE_ID)  
);  

 
CREATE TABLE RECEIPT (   
    RECEIPT_ID NUMBER GENERATED ALWAYS AS IDENTITY, 
    LINE_ID NUMBER,
    INV_DATE DATE NOT NULL,  

CONSTRAINT PK_RECEIPT_ID PRIMARY KEY(RECEIPT_ID),  
CONSTRAINT FK_LINE_ID FOREIGN KEY (LINE_ID) REFERENCES INVOICE_LINE(LINE_ID) 
);  


CREATE TABLE TAX ( 
  TAX_ID      NUMBER GENERATED ALWAYS AS IDENTITY, 
  ISO_CODE    VARCHAR2(3), 
  STATE_CODE  VARCHAR2(2), 
  TAX_RATE    NUMBER(6,4) NOT NULL, 
  EFFECTIVE_DT DATE DEFAULT SYSDATE, 

CONSTRAINT PK_TAX PRIMARY KEY (TAX_ID), 
CONSTRAINT FK_TAX_COUNTRY FOREIGN KEY (ISO_CODE) REFERENCES COUNTRY(ISO_CODE), 
CONSTRAINT FK_TAX_STATE FOREIGN KEY (STATE_CODE) REFERENCES US_STATE(STATE_CODE) 
); 

 
CREATE TABLE PAYMENT(
    PAYMENT_ID NUMBER GENERATED ALWAYS AS IDENTITY NOT NULL,   
    INVOICE_ID NUMBER NOT NULL,
    PAYMENT_AMOUNT DECIMAL(9,2),   
    PAYMENT_TYPE VARCHAR(20),   
    TAX_ID NUMBER NOT NULL,   
    RECEIPT_ID NUMBER NOT NULL,  

CONSTRAINT PK_PAYMENT_ID PRIMARY KEY (PAYMENT_ID),  
CONSTRAINT FK_PMT_INVOICE_ID FOREIGN KEY(INVOICE_ID) REFERENCES INVOICE (INVOICE_ID),   
CONSTRAINT FK_PAYMENT_TAX FOREIGN KEY (TAX_ID) REFERENCES TAX(TAX_ID), 
CONSTRAINT FK_PMT_RECEIPT_ID FOREIGN KEY(RECEIPT_ID) REFERENCES RECEIPT (RECEIPT_ID)  
);

CREATE OR REPLACE PROCEDURE sp_create_person_role 

( 

    pr_first_name          IN VARCHAR2, 

    pr_last_name           IN VARCHAR2, 

    pr_email               IN VARCHAR2, 

    pr_make_employee       IN CHAR DEFAULT 'N',  

    pr_office_id           IN NUMBER DEFAULT NULL, 

    pr_department_id       IN NUMBER DEFAULT NULL, 

    pr_make_client_contact IN CHAR DEFAULT 'N',  

    pr_client_id           IN NUMBER DEFAULT NULL, 

    pr_is_billing_contact  IN CHAR DEFAULT 'N',  

    pr_is_sales_contact    IN CHAR DEFAULT 'N', 

    pr_person_id           OUT NUMBER 

) 

AS 

    v_role CHAR(1); 

 

    -- function 1 --> Check if email exists 

     

    FUNCTION func_email_exists(pr_email_in VARCHAR2) RETURN NUMBER IS 

        v_cnt NUMBER; 

    BEGIN 

        SELECT COUNT(*) 

        INTO v_cnt 

        FROM person 

        WHERE LOWER(p_email) = LOWER(pr_email_in); 

 

        RETURN v_cnt; 

    END func_email_exists; 

 

     

    -- function 2 --> Check if client has a primary billing contact 

     

    FUNCTION func_has_primary_billing(pr_client_in NUMBER) RETURN NUMBER IS 

        v_cnt NUMBER; 

    BEGIN 

        SELECT COUNT(*) 

        INTO v_cnt 

        FROM client_contact 

        WHERE fk_client_id = pr_client_in 

          AND is_billing_contact = 'Y'; 

 

        RETURN v_cnt; 

    END func_has_primary_billing; 

 

     

    -- function 3 --> Determine person type 

     

    FUNCTION func_person_type(pr_emp CHAR, pr_cli CHAR) RETURN CHAR IS 

    BEGIN 

        IF pr_emp = 'Y' THEN 

            RETURN 'E'; 

        ELSIF pr_cli = 'Y' THEN 

            RETURN 'C'; 

        ELSE 

            RETURN 'N'; 

        END IF; 

    END func_person_type; 

 

BEGIN 

     

    -- check if unique email 

     

    IF func_email_exists(pr_email) > 0 THEN 

        RAISE_APPLICATION_ERROR(-20001, 

          'A person with this email already exists.'); 

    END IF; 

 

     

    -- find ROLE 

     

    v_role := func_person_type(pr_make_employee, pr_make_client_contact); 

 

     

    -- insert PERSON 

     

    INSERT INTO person 

        (first_name, last_name, email, person_role) 

    VALUES 

        (pr_first_name, pr_last_name, pr_email, v_role) 

    RETURNING person_id INTO pr_person_id; 

 

     

    -- insert EMPLOYEE 

     

    IF pr_make_employee = 'Y' THEN 

 

        IF pr_office_id IS NULL OR pr_department_id IS NULL THEN 

            RAISE_APPLICATION_ERROR(-20002, 

                'Employee requires office_id and department_id.'); 

        END IF; 

 

        INSERT INTO employee 

            (emp_id, person_id, office_id, department_id) 

        VALUES 

            (employee_seq.NEXTVAL, pr_person_id, pr_office_id, pr_department_id); 

 

    END IF; 

 

     

    -- insert CLIENT_CONTACT 

     

    IF pr_make_client_contact = 'Y' THEN 

 

        IF pr_client_id IS NULL THEN 

            RAISE_APPLICATION_ERROR(-20003, 

              'Client contact requires client_id.'); 

        END IF; 

 

        IF pr_is_billing_contact = 'Y' 

           AND func_has_primary_billing(pr_client_id) > 0 THEN 

            RAISE_APPLICATION_ERROR(-20004, 

              'Client already has a primary billing contact.'); 

        END IF; 

 

        INSERT INTO client_contact 

            (contact_id, fk_client_id, person_id, 

             is_billing_contact, is_sales_contact) 

        VALUES 

            (client_contact_seq.NEXTVAL, pr_client_id, pr_person_id, 

             pr_is_billing_contact, pr_is_sales_contact); 

 

    END IF; 

 

    COMMIT; 

 

END sp_create_person_role;
